export function parseToFlow(text) { const lines = text.split("\n").map(l => l.trim()).filter(Boolean); let nodes = []; let edges = []; let lastDecision = null; let lastNode = null; lines.forEach((line, i) => { const id = String(i); if (line === "START") { nodes.push({ id, type: "start", data: { label: "Start" } }); lastNode = id; return; } if (line === "END") { nodes.push({ id, type: "end", data: { label: "End" } }); edges.push({ id: `e-${lastNode}-${id}`, source: lastNode, target: id }); return; } if (line.startsWith("PROCESS:")) { const label = line.replace("PROCESS:", "").trim(); nodes.push({ id, type: "process", data: { label } }); if (lastNode) edges.push({ id: `e-${lastNode}-${id}`, source: lastNode, target: id }); lastNode = id; return; } if (line.startsWith("DECISION:")) { const label = line.replace("DECISION:", "").trim(); nodes.push({ id, type: "decision", data: { label } }); if (lastNode) edges.push({ id: `e-${lastNode}-${id}`, source: lastNode, target: id }); lastDecision = id; lastNode = id; return; } if (line.startsWith("YES:")) { const label = line.replace("YES:", "").trim(); const nid = id + "y"; nodes.push({ id: nid, type: "process", data: { label } }); edges.push({ id: `yes-${nid}`, source: lastDecision, target: nid, label: "YES" }); return; } if (line.startsWith("NO:")) { const label = line.replace("NO:", "").trim(); const nid = id + "n"; nodes.push({ id: nid, type: "process", data: { label } }); edges.push({ id: `no-${nid}`, source: lastDecision, target: nid, label: "NO" }); return; } throw new Error(`Invalid syntax: ${line}`); }); return { nodes, edges }; }
